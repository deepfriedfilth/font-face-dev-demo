Phone home (JS)

   WebFontConfig = {
      /*
      typekit: { id: 'xxxxxx' },
      google: {
        families: ['Source Sans Pro', 'Source Code Pro']
      },
      */

        families: ['GalanoGrotesque'],
        urls: ['//hello.myfonts.net/count/2fa7b0']
      }
   };

   (function(d) {
      var wf = d.createElement('script'), s = d.scripts[0];
      wf.src = 'https://ajax.googleapis.com/ajax/libs/webfont/1/webfont.js';
      wf.async = true;
      s.parentNode.insertBefore(wf, s);
   })(document);
</script>

So what did we just accomplish?

We eliminated the need for these superfluous CSS calls:

p {
    font-family: 'GalanoGrotesque-ExtraLight';
}
strong {
    font-family: 'GalanoGrotesque-Bold';
    font-weight: 400; /* no faux bold */
}
em {
    font-family: 'GalanoGrotesque-ExtraLightItalic';
    font-style: normal; /* no faux italic */
}
